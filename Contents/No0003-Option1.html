Computed

{{ upperMessage }}
    <div id="app">
        <div><input v-model="message"></div>
        <div>{{ upperMessage }}</div>
    </div>
    <script>
    var app = new Vue({
        el: '#app',
        data: {
            message: 'Hello',
        },
        computed:{
            upperMessage: function() {
                return this.message.toUpperCase();
            }
        }
    });        
    </script>

Methods

{{ message3 }}
    <div id="app">
        <button @click="toUpper">Upper</button>
        <button @click="toLower">Lower</button>
        <div>{{ message }}</div>
    </div>
    <script>
    var app = new Vue({
        el: '#app',
        data: {
            message: 'Hello world!!',
        },
        methods: {
            toUpper: function() {
                this.message = this.message.toUpperCase();
            },
            toLower: function() {
                this.message = this.message.toLowerCase();
            },
        },
    });        
    </script>

Watch

変更前：{{ oldMessage }}
変更後：{{ newMessage }}
    <div id="app">
        <form>
            <input type="text" name="text1" :value="message">
            <input type="button" @click="change" value="Change">
        </form>
        <div>変更前：{{ oldMessage }}</div>
        <div>変更後：{{ newMessage }}</div>
    </div>
    <script>
    var app = new Vue({
        el: '#app',
        data: {
            message: 'new',
            newMessage: 'new',
            oldMessage: 'old',
        },
        watch: {
            message: function(newVal, oldVal) {
                this.newMessage = newVal;
                this.oldMessage = oldVal;
            },
        },
        methods: {
            change: function(el) {
                this.message = el.target.form.text1.value;
            }
        },
    });        
    </script>

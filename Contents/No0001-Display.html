Validation

必須項目です。
文字が長すぎます。

入力した値
{{ message5 }}
    <div id="app">
        <input type="text" v-model="message" />
        <div class="error" v-if="error.require">必須項目です。</div>
        <div class="error" v-if="error.tooLong">文字が長すぎます。</div>
        <br />
        <div>入力した値</div>
        <div>{{ message }}</div>
    </div>
    <script>
    var app = new Vue({
      el: "#app",
      watch: {
        message: function (after, before) {
          this.error.require = after.length < 1 ? true : false;
          this.error.tooLong = after.length > 14 ? true : false;
        },
      },
      data: {
        message: "Hello world!!",
        error: {
          require: false,
          tooLong: false,
        },
      },
    });
    </script>

Check Box

Hello!! 
    <div id="app">
        <input type="checkbox" @click="change" checked />
        <span v-show="showMsg">Hello!!</span>
    </div>
    <script>
    var app = new Vue({
      el: "#app",
      data: {
        showMsg: true,
      },
      methods: {
        change: function (event) {
          this.showMsg = event.target.checked;
        },
      },
    });
    </script>

        </div>
    </div>
